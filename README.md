# jm-web
简盟网站

/jm-web
├── node_modules // 项目依赖包
├─┬ app //项目逻辑控制
│ ├── controllers // 各个路由
│ ├── db // 数据库定义
│ ├── utils // 工具（文件操作、日期格式、uuid、解析request中get类型参数等）
├── router // 项目路由配置
├── static // 项目静态文件和上传文件
├── tm-admin // 管理后台的打包文件
├── view // 前台页面的Ejs-html文件
├── index.js // 项目启动文件
├── package.json // 项目配置文件
